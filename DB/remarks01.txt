select 
	ef.EventFreqUid	as [Частоты контроля]
	,CPSTEF.CheckProcSeriesTaskEventFreqUid as [Частота генерации заданий ВЛК]
	,CPST.CheckProcSeriesTaskUid as [Шаблон задания серии КП]
	,CPS.CheckProcSeriesUid as [Серия контрольных процедур]
	,CPI.CheckProcInfoUid as [Информация о контрольной процедуре]
	,CP.CheckProcUid as [Контрольная процедура]
	,CPZ.CheckProcSampleUid as [Образец контрольной процедуры]
	,TSK.TaskUid as [Задание]
from EventFreq EF
inner join CheckProcSeriesTaskEventFreq CPSTEF on CPSTEF.EventFreqUid=EF.EventFreqUid
inner join CheckProcSeriesTask CPST on CPST.CheckProcSeriesTaskUid=CPSTEF.CheckProcSeriesTaskUid
inner join CheckProcSeries CPS on CPS.CheckProcSeriesUid=CPST.CheckProcSeriesUid
inner join CheckProcInfo CPI on CPI.CheckProcInfoUid=CPS.CheckProcInfoUid
inner join CheckProc CP on CP.CheckProcInfoUid=CPI.CheckProcInfoUid
inner join CheckProcSample as CPZ on CPZ.CheckProcUid=CP.CheckProcUid
inner join Task TSK on TSK.TaskUid=CPZ.TaskUid
where TSK.TaskUid='B8E94FF1-0E89-4526-A9B0-447AF6189A0C'
