            // Городим огород т.к. если читать все поля таблицы Requests возникают тормоза
            // из за докуентов. выкинем архив документов из выборки
            var _reqs = (from s in _ctx.Requests
                select new {
                    ID = s.ID
                    , DATE_DOC = s.DATE_DOC
                    , DeveloperID = s.DeveloperID
                    , DOC_Exists = s.DOC != null
                    , ErrorID = s.ErrorID
                    , MessageID = s.MessageID
                    , ProjectID = s.ProjectID
                    , StatusID = s.StatusID
                    , GUID = s.GUID
                    , packageGuid = s.packageGuid
                    , IS_IN = s.IS_IN
                    , REQ_XML = s.REQ_XML
                    , REQ_TYPE = s.REQ_TYPE
                }).ToList().Select(
                    x => new Requests()
                    {
                        ID = x.ID
                        , DATE_DOC = x.DATE_DOC
                        , DeveloperID = x.DeveloperID
                        , DOC_Exists = x.DOC_Exists
                        , ErrorID = x.ErrorID
                        , MessageID = x.MessageID
                        , ProjectID = x.ProjectID
                        , StatusID = x.StatusID
                        , GUID = x.GUID
                        , packageGuid = x.packageGuid
                        , IS_IN = x.IS_IN
                        , REQ_XML = x.REQ_XML
                        , REQ_TYPE = x.REQ_TYPE
                    }
                );
            _Requests.AddRange(_reqs);
            _AllRequests.AddRange(_reqs);
